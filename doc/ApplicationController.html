def current_region(default: [])
  return default unless params[:region]
  icao_starts = params[:region].split(/[\-,]/)
  icao_starts.compact!
  icao_starts.uniq!
  icao_starts.map!{|s| s.upcase.tr("^A-Z","")}
  return icao_starts
end
