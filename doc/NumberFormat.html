def self.coordinates(coordinates)
  return ActionController::Base.helpers.sanitize("#{"%.2f" % coordinates[0].abs}° #{coordinates[0] < 0 ? "S" : "N"}, #{"%.2f" % coordinates[1].abs}° #{coordinates[1] < 0 ? "W" : "E"}")
end

def self.date(date)
  return "" if date.nil?
  return date.strftime(DATE_FORMAT)
end
