def current_user
  return nil if cookies[:remember_token].nil?
  @current_user ||= User.find_by_remember_token(cookies[:remember_token])
end

def log_in(user)
  cookies.permanent[:remember_token] = user.remember_token
  self.current_user = user
end

def log_out

end
